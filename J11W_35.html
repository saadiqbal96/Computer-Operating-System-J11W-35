#!/usr/bin/env python3
"""
Directory Management Script (Colab Version)
Author: Saadiqbal
Description: Provides a simple menu to create, rename, or delete directories.
Designed to run in Google Colab. All operations are restricted to safe directories in /content.
"""

import os

# List of allowed directories (Colab-friendly)
ALLOWED_DIRS = [
    "/content/Projects",
    "/content/Backups",
    "/content/Shared"
]

# Ensure the allowed directories exist
for d in ALLOWED_DIRS:
    os.makedirs(d, exist_ok=True)

def list_allowed_dirs():
    print("\nAllowed directories:")
    for i, d in enumerate(ALLOWED_DIRS):
        print(f"{i+1}. {d}")

def create_directory(parent_dir, new_dir_name):
    path = os.path.join(parent_dir, new_dir_name)
    try:
        os.makedirs(path)
        print(f"✅ Directory created: {path}")
    except FileExistsError:
        print(f"⚠️ Directory already exists: {path}")

def rename_directory(parent_dir, old_name, new_name):
    old_path = os.path.join(parent_dir, old_name)
    new_path = os.path.join(parent_dir, new_name)
    try:
        os.rename(old_path, new_path)
        print(f"✅ Renamed {old_path} -> {new_path}")
    except FileNotFoundError:
        print(f"⚠️ Directory not found: {old_path}")

def delete_directory(parent_dir, dir_name):
    path = os.path.join(parent_dir, dir_name)
    try:
        os.rmdir(path)
        print(f"✅ Deleted directory: {path}")
    except FileNotFoundError:
        print(f"⚠️ Directory not found: {path}")
    except OSError:
        print(f"⚠️ Directory not empty: {path}")

def main():
    while True:
        print("\n=== Directory Management Menu ===")
        print("1. Create a New Directory")
        print("2. Rename a Directory")
        print("3. Delete a Directory")
        print("4. Exit")

        choice = input("Enter your choice (1-4): ").strip()

        if choice == "1":
            list_allowed_dirs()
            parent_index = int(input("Select parent directory by number: ")) - 1
            new_dir_name = input("Enter new directory name: ").strip()
            create_directory(ALLOWED_DIRS[parent_index], new_dir_name)

        elif choice == "2":
            list_allowed_dirs()
            parent_index = int(input("Select parent directory by number: ")) - 1
            old_name = input("Enter existing directory name: ").strip()
            new_name = input("Enter new directory name: ").strip()
            rename_directory(ALLOWED_DIRS[parent_index], old_name, new_name)

        elif choice == "3":
            list_allowed_dirs()
            parent_index = int(input("Select parent directory by number: ")) - 1
            dir_name = input("Enter directory name to delete: ").strip()
            delete_directory(ALLOWED_DIRS[parent_index], dir_name)

        elif choice == "4":
            print("Exiting script. ✅")
            break

        else:
            print("⚠️ Invalid choice. Try again.")

if __name__ == "__main__":
    main()
